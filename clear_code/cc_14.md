```
    ind = -1
    for i in range(len(list1)):
        if list1[i] == n:
            ind = i
    return ind
```

```
def FindNumber(numbers: list, target: int) -> int:
    index = 0
    for number in numbers:
        if number == target:
            return index
        index += 1
    return -1
```

```
import re

def white_walkers(village: str) -> bool:
    safe = False
    counter = 0
    first_d = 0
    if len(village) < 5:
        return False
    for i in range(len(village)):
        if village[i] == '=':
            counter += 1
            continue
        if re.match(r"\d", village[i]):
            if int(first_d) + int(village[i]) == 10:
                if counter == 3:
                    safe = True
                else:
                    safe = False
            first_d = village[i]
            counter = 0
            continue
        else:
            continue
    return safe
```

```    
def MaximumDiscount(N: int, price: list) -> int:

    price.sort(reverse=True)
    bulk = []
    summ = 0
    for i in range(N):
        bulk.append(price[i])
        if len(bulk) % 3 == 0:
            summ += min(bulk)
            bulk = []

    return summ
```
