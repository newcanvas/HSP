```
# Рекурсионная сортировка массива.
def TRC_sort_rec(arr, i, zeros, ones, twos):
    if i == len(arr):
        return zeros + ones + twos

    if arr[i] == 0:
        return TRC_sort_rec(arr, i+1, zeros+[0], ones, twos)
    if arr[i] == 1:
        return TRC_sort_rec(arr, i+1, zeros, ones+[1], twos)
    if arr[i] == 2:
        return TRC_sort_rec(arr, i+1, zeros, ones, twos+[2])
```

```
def squirrel(N: int) -> int:
    F = 1
    for i in range(2, N+1):
        F = F * i

    first_d = int(str(F)[0])
```

```
# Функция, находящая сумму максимальной скидки при покупках.
def MaximumDiscount(N: int, price: list) -> int:

    price.sort(reverse=True)
    bulk = []
    summ = 0
    for purchase in price:
        bulk.append(purchase)
        if len(bulk) % 3 == 0:
            summ += min(bulk)
            bulk = []

    return summ
```

```
# Функция, находящая индекс числа в массиве.
def FindNumber(numbers: list, target: int) -> int:
    index = 0
    for number in numbers:
        if number == target:
            return index
        index += 1
    return -1
```

```
# Функция, вычисляющая километраж поездки.
def odometer(oksana: list) -> int:
    total = 0
    for i in range(len(oksana)):
        if i == 0:
            hours = oksana[i+1]
            total += oksana[i] * hours
        elif i % 2 == 0:
            hours = oksana[i+1] - oksana[i-1]
            total += oksana[i] * hours
    assert total > -1
    return total
```

```
# Функция, упорядовачивающая список зарплат в массиве.
def SynchronizingTables(N: int, ids: list, salary: list):

    ids_sorted = sorted(ids.copy())
    salary_sorted = sorted(salary.copy())

    for i in range(N):
        salary[ids.index(ids_sorted[i])] = salary_sorted[i]

    return salary
```
