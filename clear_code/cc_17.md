```
# Функция, возвращающая первую цифру факториала N!
def squirrel(N: int) -> int:
    F = 1
    for i in range(2, N+1):
        F = F * i

    first_d = int(str(F)[0])
    return first_d
```

```
# Рекурсионная сортировка массива.
def TRC_sort_rec(arr, i, zeros, ones, twos):
    if i == len(arr):
        return zeros + ones + twos

    if arr[i] == 0:
        return TRC_sort_rec(arr, i+1, zeros+[0], ones, twos)
    if arr[i] == 1:
        return TRC_sort_rec(arr, i+1, zeros, ones+[1], twos)
    if arr[i] == 2:
        return TRC_sort_rec(arr, i+1, zeros, ones, twos+[2])
```

```
# Функция, находящая индекс числа в массиве.
def FindNumber(numbers: list, target: int) -> int:
    index = 0
    for number in numbers:
        if number == target:
            return index
        index += 1
    return -1
```
